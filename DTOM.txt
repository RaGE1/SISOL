#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int a[n][n]={0};
        int a1[2*n-1]={0};
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                cin>>a[i][j];
                if(j>=i)
                {
                    a1[j-i]+=a[i][j];
                    //cout<<"a1["<<j<<"-"<<i<<"]"<<a1[j-i]<<" "<<a[i][j]<<endl;
                }
                else{
                    //cout<<a1[i-j+n];
                    a1[i-j+n-1]+=a[i][j];
                    //cout<<"a1["<<i<<"-"<<j<<"]"<<a1[i-j+n]<<" "<<a[i][j]<<endl;
                }
            }
        }
        for(int i=n-1;i>=0;i--)
            cout<<a1[i]<<" ";
        for(int i=n;i<2*n-1;i++)
            cout<<a1[i]<<" ";
        
        cout<<endl;
    }
    return 0;
}
